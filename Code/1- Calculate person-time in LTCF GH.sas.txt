*Program name: 1-Calculate person-time in LTCF*Last updated date: May 3, 2026*Purpose: Create a final dataset with residents who were in house during the Pre- and Post- STRIDE intervetion periods,alongwith corrected overlaps and person-time.Note: In this final dataset, a person can only be repeated more than once if they were in house during both pre- and post-STRIDE intervention periods;libname ltcdc "YOUR PATH";libname output "YOUR PATH";*Subset to dates of the STRIDE intervention and patient status="in house";data output.stay_base;	set output.stay;	start_date_created=datepart(start_date);	end_date_created=datepart(end_date);	where end_date~=.;	if "18jan2022"d<=start_date_created<="17jan2024"d or "18jan2022"d<=end_date_created<="17jan2024"d or (start_date_created<"18jan2022"d and end_date_created>"17jan2024"d);	if patient_status="In House";run;*Flag stays as pre, post or both pre and post STRIDE;data output.flag_periods;	length period_1 period_2 $20;	set output.stay_base;	if "18jan2022"d<=start_date_created<="17jan2023"d or "18jan2022"d<=end_date_created<="17jan2023"d or (start_date_created<"18jan2022"d and end_date_created>"17jan2023"d)	then period_1="Pre-STRIDE";	else period_1="Blank";	if "18jan2023"d<=start_date_created<="17jan2024"d or "18jan2023"d<=end_date_created<="17jan2024"d or (start_date_created<"18jan2023"d and end_date_created>"17jan2024"d)	then period_2="Post-STRIDE";	else period_2="Blank";	if period_1="Pre-STRIDE" and period_2="Post-STRIDE" then both_period=1;else both_period=0;run;*Correct overlapping stays;proc sort data=output.flag_periods;	by master_patient_id start_date_created end_date_created;run;data new;	set output.flag_periods;	by master_patient_id;	retain new_end_date;	if first.master_patient_id then new_end_date =end_date_created;	else new_end_date = max(end_date_created, new_end_date); 	 lag_new_end_date= lag(new_end_date); 	 if first.master_patient_id then do;	 episode = 1;	 lag_new_end_date="18jan2022"d ;	 end;/*	 *if the maximum discontinue date across episodes is less than the current value of the *//*	 fill date, then flag this dispensing as part of a new medication episode;*/	 *otherwise flag as the same episode;	 else if lag_new_end_date < start_date_created then episode+1; 	 else episode=episode;	 format  end_date_created start_date_created lag_new_end_date new_end_date date9.;run;data interim;	set new;	keep master_patient_id episode end_date_created start_date_created lag_new_end_date new_end_date start_reason end_reason;run;proc freq data=interim;	tables lag_new_end_date new_end_date;run;proc sql;	 create table distinct_episodes as	 select distinct master_patient_id, episode, /*	 *Take the minimum start date of all overlapping medication episodes;*/	 min(start_date_created) as min_start_date format=date9., /*	 *Take the maximum supply end date of all overlapping medication episodes;*/	 max(end_date_created) as max_end_date format=date9./*	 calculated max_end_date- calculated min_start_date+1 as days_stay*/	 from new	 group by master_patient_id, episode	 order by master_patient_id, episode, min_start_date, max_end_date;quit;data episode_flag;	set distinct_episodes;	if "18jan2022"d<=min_start_date<="17jan2023"d or "18jan2022"d<=max_end_date<="17jan2023"d or (min_start_date<"18jan2022"d and max_end_date>"17jan2023"d)	then period_1="Pre-STRIDE";	else period_1="Blank";	if "18jan2023"d<=min_start_date<="17jan2024"d or "18jan2023"d<=max_end_date<="17jan2024"d or (min_start_date<"18jan2023"d and max_end_date>"17jan2024"d)	then period_2="Post-STRIDE";	else period_2="Blank";	if period_1="Pre-STRIDE" and period_2="Post-STRIDE" then both_period=1;else both_period=0;run;*Work with individuals who appeared in both periods;data both_period;	set episode_flag;	if both_period=1;run;proc sort data=both_period;	by master_patient_id;run;data duplicate_observation;	length period_master $14;	set both_period (in=a) both_period (in=b);	if a=1 then period_master="Pre-STRIDE";	else period_master="Post-STRIDE";	if period_master="Post-STRIDE" then start_date_altered="18jan2023"d;	else start_date_altered=min_start_date;	if period_master="Pre-STRIDE" then end_date_altered="17jan2023"d;	else end_date_altered=max_end_date;	format end_date_altered start_date_altered date9.;run;proc sort data=duplicate_observation;	by master_patient_id;run;*Work with stays in one period only;data one_period;	length period_master $14;	set episode_flag;	if both_period=0;	if period_1="Pre-STRIDE" then period_master="Pre-STRIDE";	else if period_2="Post-STRIDE" then period_master="Post-STRIDE";run;proc freq data=one_period;	tables period_master;run;data corrected_stays;	set duplicate_observation one_period(rename=( min_start_date=start_date_altered max_end_date=end_date_altered));	keep master_patient_id start_date_altered end_date_altered period_master;run;proc freq data=corrected_stays;	tables period_master;run;proc sort data=corrected_stays;	by master_patient_id;run;proc freq data=corrected_stays;	tables start_date_altered;	where period_master="Pre-STRIDE";run;*Start date needs to be set to "18jan2022"d;proc freq data=corrected_stays;	tables start_date_altered;	where period_master="Post-STRIDE";run;*start dates post-stride all ok, no need to adjust;proc freq data=corrected_stays;	tables end_date_altered;	where period_master="Pre-STRIDE";run;*all ok;proc freq data=corrected_stays;	tables end_date_altered;	where period_master="Post-STRIDE";run;*end date goes beyong stride, and missing;data stays_final;	set corrected_stays;	if period_master="Pre-STRIDE" and start_date_altered<"18jan2022"d then stay_start="18jan2022"d;	else stay_start=start_date_altered;	if period_master="Post-STRIDE" then stay_end=min(end_date_altered, "17jan2024"d);	else stay_end=end_date_altered;	days_stay_final=stay_end-stay_start+1;run;proc sql;	create table output.distinct_person_time as	select master_patient_id, sum(days_stay_final) as days_stay_total, period_master, count(*) as count_stays	from stays_final	group by master_patient_id, period_master;quit;proc means data=output.distinct_person_time n nmiss;	var days_stay_total;run;proc sql;	create table check_recs as	select distinct master_patient_id, count(*) as count_recs	from output.distinct_person_time	group by master_patient_id;quit;proc freq data=check_recs;	tables count_recs;run;
